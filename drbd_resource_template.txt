resource $drbd_res_name { # имя ресурса
	startup {
	wfc-timeout 120;
	degr-wfc-timeout 160;
	}
	net {
	cram-hmac-alg sha1;
	shared-secret "12345678";
	after-sb-0pri discard-zero-changes;
	after-sb-1pri discard-secondary;
	after-sb-2pri disconnect;
	}
	syncer {
	verify-alg sha1;
	rate 30M;
	}
	on sbzd1.nicetu {# имя ноды 1
	device /dev/drbd$station_number; #раздел drbd, используемый в качестве таргета
	disk /dev/mapper/$lvm_volume_name; #используемый том LVM соответствующего клиента
	address $server1_ip_address:7788; # сетевой адрес ноды и порт синхронизируемого блочного устройства
	meta-disk internal;
	}
	on sbdz2.nicetu {# имя ноды 2
	device /dev/drbd$station_number; #раздел drbd, используемый в качестве таргета
	disk /dev/mapper/$lvm_volume_name; #используемый том LVM соответствующего клиента
	address $server2_ip_address:7788; # сетевой адрес ноды и порт синхронизируемого блочного устройства
	meta-disk internal;
	}
	}
